Plan <- read.table("http://www.agrocampus-ouest.fr/math/husson/StatUtilisateurExo/chap14_ex5.csv", header = TRUE, sep=";",row.names=1)

X <- model.matrix(~ X1+X2+X3+I(X1^2)+I(X2^2)+I(X3^2)+I(X1*X2)+I(X2*X3)+I(X1*X3), Plan)
round(X,3)
round(cor(X),3)
round(solve(t(X)%*%X),3)

## Plan avec 3 points au centre
Plan2 <- Plan[c(1,1,1:13),]

round(X,3)
