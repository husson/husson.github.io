Plan <- read.table("http://www.agrocampus-ouest.fr/math/husson/StatUtilisateurExo/chap14_ex8.csv", header = TRUE, sep=";",row.names=1)

X <- model.matrix(~ F1+F2+F3+F4+F5+I(F1^2)+I(F2^2)+I(F3^2)+I(F4^2)+I(F5^2), Plan)
round(X,3)
round(cor(X),3)
round(solve(t(X)%*%X),3)
