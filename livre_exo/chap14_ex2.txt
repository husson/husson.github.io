Plan <- read.table("http://www.agrocampus-ouest.fr/math/husson/StatUtilisateurExo/chap14_ex2.csv", header = TRUE, sep=";",row.names=1)

X <- model.matrix(~ X1+X2+I(X1*X2)+I(X1^2)+I(X2^2), Plan)
t(X)%*%X
cor(X)
solve(t(X)%*%X)

res <- lm(Y ~ X1 + X2 + I(X1^2) + I(X2^2) + I(X1*X2), data = Plan)
summary(res)
round(residuals(res),3)
